#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
#define all(v) v.begin(), v.end()
#define pb push_back

int bin_search(vector<int>& a, int res) {
	int l = 0;
	int r = a.size() - 1;

	while (l <= r) {
		int m = l + (r - l) / 2;
		if (a[m] == res) {
			return m + 1;
		}
		else if (a[m] < res) {
			l = m + 1;
		}
		else {
			r = m - 1;
		}
	}
	return 0;
}

int n, m, q;
int a, b;

int main() {
	cin >> n >> m;
	vector<vector<int>> graph(n + 1);
	for (int i = 0; i < m; i++) {
		cin >> a >> b;
		graph[a].pb(b);
		graph[b].pb(a);
	}
	for (int i = 1; i < n + 1; i++) {
		sort(all(graph[i]));
	}
	cin >> q;
	for (int i = 0; i < q; i++) {
		cin >> a >> b;
		cout << bin_search(graph[a], b) << endl;
	}
}
