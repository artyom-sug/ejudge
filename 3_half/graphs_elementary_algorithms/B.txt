#include <iostream>
#include <vector>
using namespace std;
#define pub push_back
#define pob pop_back

int n, m, x, y, cnt = 0;
bool flag;

int main() {
	cin >> n >> m;
	vector<vector<int>> graph(n + 1);
	for (int i = 0; i < m; i++) {
		cin >> x >> y;
		graph[x].pub(y);
		graph[y].pub(x);
	}
	for (int i = 1; i < n + 1; i++) {
		if (graph[i].empty()) continue;
		flag = 1;
		for (int j = 0; j < graph[i].size(); j++) {
			if (graph[graph[i][j]].size() != 1) flag = 0;
			if (graph[graph[i][j]].size() == 1 and graph[i].size() == 1) graph[graph[i][j]].pob();
			if (!flag) break;
		}
		if (flag) cnt++;
	}
	cout << cnt;
}
