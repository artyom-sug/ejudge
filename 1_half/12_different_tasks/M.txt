#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int	a, b, n;

int main() {
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> a >> b;
		cout << (b - a) + 1 << endl;
		for (int i = a; i <= b; i++) { 
			if (i != b) cout << i << ' ';
			else cout << i << endl;
		}
	}
}
