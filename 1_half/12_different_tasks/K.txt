#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

long long t, a, b;
long double c, p, s;

int main() {
	cin >> t;
	for (int i = 0; i < t; i++) {
		cin >> a >> b;
		c = sqrt(a * a + b * b);
		p = a + b + c;
		s = 0.5 * b * a;
		cout <<fixed<<setprecision(3)<< p << " " << s << endl;
	}
}
